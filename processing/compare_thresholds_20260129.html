py_arrays = []

for thr in thresholds:
    thr_str = f"{thr:.1f}".replace(".", "p")
    subdir = f"output_thr-{thr_str}"
    path = base_dir / "py" / subdir / f"TM{timepoint:06d}" / f"SPM00_TM{timepoint:06d}_CM{camera:02d}_CHN{channel:02d}.klb"
    
    if path.exists():
        arr = np.asarray(imread(path))
        arr = np.swapaxes(arr, -1, -2)  # swap Y and X to match matlab orientation
        py_arrays.append(arr)
        py_names.append(f"py thr={thr}")
        print(f"Loaded {path.name}: shape={arr.shape}")
    else:
        print(f"Not found: {path}")

    vmax = max(abs(g.data[:].min()), abs(g.data[:].max()))
    g.vmin = -vmax
    g.vmax = vmax

for subplot in iw_diff.figure:
    subplot.toolbar = False

import matplotlib.pyplot as plt

fig, axes = plt.subplots(2, len(diff_samples), figsize=(4 * len(diff_samples), 6))

for i, (diff, thr) in enumerate(zip(diff_samples, thresholds)):
    diff_flat = diff.ravel()
    nonzero = diff_flat[diff_flat != 0]
    
    # histogram
    ax_hist = axes[0, i]
    if len(nonzero) > 0:
        ax_hist.hist(nonzero, bins=50, color='steelblue', edgecolor='black')
        ax_hist.set_title(f"thr={thr}\n{len(nonzero)} nonzero ({100*len(nonzero)/len(diff_flat):.2f}%)")
    else:
        ax_hist.text(0.5, 0.5, "No differences", ha='center', va='center', transform=ax_hist.transAxes)
        ax_hist.set_title(f"thr={thr}\n0 nonzero (0%)")
    ax_hist.set_xlabel("diff value")
    ax_hist.set_ylabel("count")
    
    # stats text
    ax_stats = axes[1, i]
    ax_stats.axis('off')
    stats_text = (
        f"min: {diff.min():.2f}\n"
        f"max: {diff.max():.2f}\n"
        f"mean: {diff.mean():.4f}\n"
        f"std: {diff.std():.4f}\n"
        f"abs mean: {np.abs(diff).mean():.4f}\n"
        f"nonzero: {len(nonzero)}/{len(diff_flat)}"
    )
    ax_stats.text(0.5, 0.5, stats_text, ha='center', va='center', transform=ax_stats.transAxes,
                  fontsize=11, family='monospace')

plt.tight_layout()
plt.show()
