masknmf-toolbox¶

Complete workflow: data loading → motion correction → compression/denoising → demixing → visualization

masknmf-toolbox is a PyTorch implementation for functional neuroimaging analysis (calcium, voltage, glutamate imaging). It provides GPU-accelerated pipelines for motion correction, compression/denoising, signal demixing, and visualization.

Pipeline Steps¶

The pipeline processes imaging data through these stages:

  1. arrays/ - data loading interfaces for .tiff and .hdf5 formats

    • TiffArray, Hdf5Array - format-specific loaders
    • LazyFrameLoader, FactorizedVideo - lazy loading abstractions

  2. motion_correction/ - GPU-accelerated registration

    • RigidMotionCorrector, PiecewiseRigidMotionCorrector - main strategies
    • RegistrationArray - lazy array wrapper for registered frames
    • registration_methods.py - core registration algorithms using PyTorch FFT

  3. compression/ - PMD (penalized matrix decomposition) compression

    • PMDArray, PMDResidualArray - compression containers
    • denoising.py - temporal denoising with total variation
    • decomposition.py - low-rank SVD factorization

  4. demixing/ - neural source separation

    • SignalDemixer - main orchestrator
    • ACArray - activity correlation arrays
    • RingModel - background estimation
    • graph-based and regression-based algorithms for signal extraction

  5. visualization/ - interactive tools using fastplotlib

    • PMDWidget - compression visualization
    • interactive_guis.py - demixing GUI

  6. diagnostics/ - quality assessment utilities

See ibl_scripts/ for the 3-step IBL processing pipeline:

  1. motion_correct_data.py - registration
  2. compress_and_denoise.py - PMD compression
  3. demix_data.py - signal extraction

Dependencies¶

Core: PyTorch >= 2.0, numpy, scipy, h5py, cvxpy, pytorch-lightning, fastplotlib, networkx

class PlaneView:
    """lazy single-plane view into 4D array (T, Z, H, W) -> (T, H, W)"""
    def __init__(self, data, plane_idx):
        self._data = data
        self._plane = plane_idx
        self.ndim = 4
        
    @property
    def shape(self):
        return (self._data.shape[0], self._data.shape[2], self._data.shape[3])
    
    @property
    def dtype(self):
        return self._data.dtype
    
    def __getitem__(self, key):
        if isinstance(key, int):
            return self._data[key, self._plane]
        if isinstance(key, slice):
            return self._data[key, self._plane]
        if isinstance(key, tuple):
            t_key = key[0] if len(key) > 0 else slice(None)
            y_key = key[1] if len(key) > 1 else slice(None)
            x_key = key[2] if len(key) > 2 else slice(None)
            return self._data[t_key, self._plane, y_key, x_key]
        return self._data[key, self._plane]
    
    def __len__(self):
        return self._data.shape[0]

def normalize_for_display(arr, vmin=None, vmax=None):
    """normalize array to 0-1 range for display"""
    if vmin is None:
        vmin = np.percentile(arr, 1)
    if vmax is None:
        vmax = np.percentile(arr, 99)
    arr = np.clip(arr, vmin, vmax)
    arr = (arr - vmin) / (vmax - vmin + 1e-8)
    return arr

def process_plane(raw_data, plane_idx, output_base, device="cuda"):
"""process a single plane through the full pipeline"""
from masknmf.compression.denoising import train_total_variance_denoiser, PMDTemporalDenoiser
from masknmf.demixing.filters import spatial_filter_pmd
from masknmf.demixing import SignalDemixer


# extract plane
if raw_data.ndim == 4:
    data = PlaneView(raw_data, plane_idx)
else:
    data = raw_data

plane_dir = output_base / f"plane{plane_idx:02d}"
plane_dir.mkdir(parents=True, exist_ok=True)

print(f"\n=== plane {plane_idx}: {data.shape} ===")

# motion correction
moco_path = plane_dir / "motion_corrected.h5"
moco_strategy = masknmf.PiecewiseRigidMotionCorrector(
    num_blocks=(10, 10),
    overlaps=(5, 5),
    max_rigid_shifts=(15, 15),
    max_deviation_rigid=(2, 2),
    batch_size=500,
    device=device,
)
moco_strategy.compute_template(data)
moco_data = masknmf.RegistrationArray(data, moco_strategy)
moco_data.export(moco_path)
print(f"  moco done")

# compression
pmd_path = plane_dir / "compressed.h5"
moco_data = masknmf.Hdf5Array(moco_path, field="motion_corrected")
h, w = moco_data.shape[1], moco_data.shape[2]
pw = np.zeros((h, w), dtype=np.float32)
pw[3:-3, 3:-3] = 1.0

cs = masknmf.CompressStrategy(moco_data, block_sizes=[32, 32], max_components=20,
                               frame_batch_size=1024, pixel_weighting=pw, device=device)
pmd_raw = cs.compress()

v = pmd_raw.v.cpu()
dm = train_total_variance_denoiser(v, max_epochs=5, batch_size=128, device=device)
td = PMDTemporalDenoiser(dm, noise_variance_quantile=0.7)

cds = masknmf.CompressDenoiseStrategy(moco_data, temporal_denoiser=td, block_sizes=[32, 32],
                                       max_components=20, frame_batch_size=1024,
                                       pixel_weighting=pw, device=device)
pmd_data = cds.compress()
pmd_data.export(pmd_path)
print(f"  compression done")

# demixing
demix_path = plane_dir / "demixing_results.h5"
pmd_data = masknmf.PMDArray.from_hdf5(pmd_path)

def process_plane(raw_data, plane_idx, output_base, device="cuda"):
    """process a single plane through the full pipeline"""
    from masknmf.compression.denoising import train_total_variance_denoiser, PMDTemporalDenoiser
    from masknmf.demixing.filters import spatial_filter_pmd
    from masknmf.demixing import SignalDemixer
    
    # extract plane
    if raw_data.ndim == 4:
        data = PlaneView(raw_data, plane_idx)
    else:
        data = raw_data
    
    plane_dir = output_base / f"plane{plane_idx:02d}"
    plane_dir.mkdir(parents=True, exist_ok=True)
    
    print(f"\n=== plane {plane_idx}: {data.shape} ===")
    
    # motion correction
    moco_path = plane_dir / "motion_corrected.h5"
    moco_strategy = masknmf.PiecewiseRigidMotionCorrector(
        data, num_blocks=[10, 10], overlaps=[5, 5],
        max_rigid_shifts=[15, 15], max_deviation_rigid=[2, 2],
        batch_size=500, device=device,
    )
    moco_strategy.compute_template(data)
    moco_data = masknmf.RegistrationArray(data, moco_strategy)
    moco_data.export(moco_path)
    print(f"  moco done")
    
    # compression
    pmd_path = plane_dir / "compressed.h5"
    moco_data = masknmf.Hdf5Array(moco_path)
    h, w = moco_data.shape[1], moco_data.shape[2]
    pw = np.zeros((h, w), dtype=np.float32)
    pw[3:-3, 3:-3] = 1.0
    
    cs = masknmf.CompressStrategy(moco_data, block_sizes=[32, 32], max_components=20,
                                   frame_batch_size=1024, pixel_weighting=pw, device=device)
    pmd_raw = cs.compress()
    
    v = pmd_raw.v.cpu()
    dm = train_total_variance_denoiser(v, max_epochs=5, batch_size=128, device=device)
    td = PMDTemporalDenoiser(dm, noise_variance_quantile=0.7)
    
    cds = masknmf.CompressDenoiseStrategy(moco_data, temporal_denoiser=td, block_sizes=[32, 32],
                                           max_components=20, frame_batch_size=1024,
                                           pixel_weighting=pw, device=device)
    pmd_data = cds.compress()
    pmd_data.export(pmd_path)
    print(f"  compression done")
    
    # demixing
    demix_path = plane_dir / "demixing_results.h5"
    pmd_data = masknmf.PMDArray.from_hdf5(pmd_path)
    
    pmd_filt = spatial_filter_pmd(pmd_data, filter_sigma=(4, 4))
    di = SignalDemixer(pmd_filt, device=device)
    di.initialize_signals(mad_correlation_threshold=0.8, mad_threshold=1,
                          residual_threshold=0.3, patch_size=(40, 40), is_custom=False)
    di.demix(maxiter=25, support_threshold=np.linspace(0.95, 0.7, 25).tolist(),
             deletion_threshold=0.2, ring_model_start_pt=26, merge_threshold=0.6,
             merge_overlap_threshold=0.3, c_nonneg=True, denoise=False)
    
    d = SignalDemixer(pmd_data, device=device)
    d.initialize_signals(spatial_footprints=di.get_spatial_footprints(), is_custom=True)
    d.demix(maxiter=25, support_threshold=np.linspace(0.95, 0.5, 25).tolist(),
            deletion_threshold=0.2, ring_model_start_pt=0, ring_radius=15,
            background_downsampling_factor=20, merge_threshold=0.8,
            merge_overlap_threshold=0.8, c_nonneg=True, denoise=False)
    
    results = d.get_results()
    results.export(demix_path)
    print(f"  demixing done: {results.num_neurons} neurons")
    
    return results
